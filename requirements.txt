# FBD-WSLGUI Python Dependencies
# =====================================
# 
# SYSTEM PACKAGES (install via apt/dnf/pacman):
# - python3-tk      : GUI framework (tkinter)
# - python3-requests: HTTP library for RPC calls
# - python3-pip      : pip installer (for customtkinter)
#
# Ubuntu/Debian:
#   sudo apt update && sudo apt install -y python3-tk python3-requests
#
# Fedora/RHEL/CentOS:
#   sudo dnf install -y python3-tkinter python3-requests
#
# Arch Linux:
#   sudo pacman -S tk python-requests
#
# =====================================
# PIP ALTERNATIVE (if system packages unavailable):
# =====================================
# Note: tkinter usually can't be pip-installed - use system package manager
# Python package dependencies installed via pip:

requests>=2.25.0
customtkinter>=5.2.0
